{{% alert color="primary" %}}
Чтобы интегрировать Aspose.CAD для JasperReports с JasperServer, необходимо выполнить несколько дополнительных шагов и обновить конфигурационные файлы JasperServer. Эта статья объясняет, как это сделать.
{{% /alert %}}

    Чтобы изменить режим экспорта, используйте свойство с именем "exportMode" и установите значение в соответствии с желаемым режимом. 
Например:
{{< highlight xml >}}
    <property name="exportMode">
        <value type="com.aspose.cad.jasperreports.common.ExportMode">SinglePage</value>
    </property>
{{< /highlight >}}
    Вы можете изменить значение свойства на Batch, SinglePage или Multipage, используя перечисление com.aspose.cad.jasperreports.common.ExportMode.

2. Найдите элемент <util:map id="exporterConfigMap">...</util:map> в файле %INTALL_DIR%\\apache-tomcat\webapps\jasperserver\WEB-INF\flows\viewReportBeans.xml и добавьте следующие строки:
{{< highlight xml >}}
    <util:map id="exporterConfigMap">
        <!-- закомментируйте/раскомментируйте любые строки ниже, если хотите исключить/включить соответствующих экспортеров 
        	 в список экспортеров просмотрщика
        	 Примечание: отдельная конфигурация для iPad 'exportersSupportedByiPad'
        	 -->
        	 
        <entry key="pdf" value-ref="pdfExporterConfiguration"/>
        <entry key="xls" value-ref="xlsExporterConfiguration"/>
        <entry key="xlsNoPag" value-ref="xlsNoPaginationExporterConfiguration"/>
        <entry key="csv" value-ref="csvExporterConfiguration"/>
        <entry key="docx" value-ref="docxExporterConfiguration"/>
        <entry key="rtf" value-ref="rtfExporterConfiguration"/>
        <entry key="odt" value-ref="odtExporterConfiguration"/>
        <entry key="ods" value-ref="odsExporterConfiguration"/>
        <entry key="xlsx" value-ref="xlsxExporterConfiguration"/>
        <entry key="xlsxNoPag" value-ref="xlsxNoPaginationExporterConfiguration"/>
        <entry key="pptx" value-ref="pptxExporterConfiguration"/>
        <!-- 
        <entry key="txt" value-ref="txtExporterConfiguration"/>
        -->
		
			<!-- добавьте эти записи в exporterConfigMap -->
			<!-- Aspose.CAD JasperReports НАЧАЛО -->
		<entry key="as_jpg" value-ref="jpgASExporterParameters"/>
		<entry key="as_bmp" value-ref="bmpASExporterParameters"/>
		<entry key="as_gif" value-ref="gifASExporterParameters"/>
		<entry key="as_jpg2000" value-ref="jpg2000ASExporterParameters"/>
		<entry key="as_pdf" value-ref="pdfASExporterParameters"/>
		<entry key="as_png" value-ref="pngASExporterParameters"/>
		<entry key="as_psd" value-ref="psdASExporterParameters"/>
		<entry key="as_svg" value-ref="svgASExporterParameters"/>
		<entry key="as_tiff" value-ref="tiffASExporterParameters"/>
		<entry key="as_wmf" value-ref="wmfASExporterParameters"/>
		<!-- Aspose.CAD для JasperReports КОНЕЦ -->
    </util:map>
{{< /highlight >}}

![todo:image_alt_text](/_assets/jasper/ExportReportView.png)

Если вы не видите дополнительных форматов (значков или вариантов), пожалуйста, проверьте журналы в каталоге \apache-tomcat\logs.
